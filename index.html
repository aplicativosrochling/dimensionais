Preferências

☁ Nuvem (Supabase) — login e sincronização

Conecte um projeto Supabase para acessar seus relatórios de qualquer dispositivo, com login por e-mail e senha. É opcional — sem isso, tudo continua salvo só neste navegador.

Ver script SQL de configuração (rodar 1x no SQL Editor do Supabase) 
create table if not exists public.dimensional_reports (
  id text primary key,
  user_id uuid not null default auth.uid() references auth.users(id) on delete cascade,
  name text not null,
  data jsonb not null,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

alter table public.dimensional_reports enable row level security;

create policy "Users manage only their own reports"
on public.dimensional_reports
for all
using (auth.uid() = user_id)
with check (auth.uid() = user_id);

-- Immutable version history: every Excel/JSON export inserts a new row here (never overwritten)
create table if not exists public.dimensional_report_versions (
  id uuid primary key default gen_random_uuid(),
  report_id text not null,
  user_id uuid not null default auth.uid() references auth.users(id) on delete cascade,
  version int not null,
  name text not null,
  data jsonb not null,
  created_at timestamptz not null default now()
);

alter table public.dimensional_report_versions enable row level security;

create policy "Users manage only their own report versions"
on public.dimensional_report_versions
for all
using (auth.uid() = user_id)
with check (auth.uid() = user_id);
